Struct std::collections::HashSet

Re-export from std::collections::hashset::HashSet

struct HashSet<K, H = DefaultHash> { ... }
K: Hashable<K, H> + Equatable<K>
H: Hasher<H>

source

A hash-based set collection.

HashSet<T> is a wrapper around HashMap<T, ()>.

Example

use std::collections::HashSet;

let map = HashSet::new::<i32>();
defer map.free();

map.insert(1);
map.insert(2);
map.insert(3);

assert!(map.contains(1));
assert!(map.contains(2));
assert!(map.contains(3));

    source

    Creates a hash map from an iterator of key-value pairs

    Example

    use std::collections::HashSet;
let set: HashSet<i32> = HashSet::from_iter(&[1, 2, 3].iter());
defer set.free();

assert_eq!(set.len(), 3);
assert!(set.contains(1));
assert!(set.contains(2));
assert!(set.contains(3));
