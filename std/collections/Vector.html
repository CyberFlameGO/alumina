Struct std::collections::Vector

struct Vector<T> { ... }

source

Growable heap-allocated array.

Example

let vec = Vector::new::<i32>();
defer vec.free();

vec.push(1);
assert_eq!(vec.len(), 1);
assert_eq!(vec.as_slice(), &[1]);
vec.push(2);
assert_eq!(vec.len(), 2);
assert_eq!(vec.as_slice(), &[1, 2]);
vec.push(3);
assert_eq!(vec.len(), 3);
assert_eq!(vec.as_slice(), &[1, 2, 3]);
