Macro std::fmt::format

macro format($fmt_str, $arg...) { ... }

source

Format a string into a StringBuf.

Example

 use std::fmt::format;

 let s: StringBuf = format!("{} + {} = {}", 1, 2, 3);
 defer s.free()

 assert_eq!(s.as_slice(), "1 + 2 = 3");
