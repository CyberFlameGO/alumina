struct DirBuilder { ... }

source

Utility for creating directories.

Example

use std::fs::{DirBuilder, Path};

DirBuilder::new()
    .recursive(true)
    .create(Path::new("/home/alumina/a/b/c/d/e/f"))
    .unwrap();
