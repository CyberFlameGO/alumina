  • fn read_directory(path: Path) -> Result<DirIterator>

    source

    Returns iterator over the items in a directory.

    Example

    use std::fs::read_directory;

let dir = read_directory("/home/alumina")?
defer dir.close()?;

for item in dir {
    let item = item?;
    println!("{}: {}", item.name(), item.file_type());
}
