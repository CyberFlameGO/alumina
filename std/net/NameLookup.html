struct NameLookup { ... }

source

Domain name resolution iterator

Example

use std::net::NameLookup;
let addresses = NameLookup::resolve("www.alumina-lang.net", 443).unwrap();
defer addresses.free();

for addr in addresses {
    println!("{}", addr);
}
